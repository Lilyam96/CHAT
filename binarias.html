// @version=6 indicator("MASTER V11.1: SNIPER DUAL DISPLAY", overlay=true, max_labels_count=500, max_boxes_count=500) // --- INPUTS --- lbP = input.int(8, "Micro-Estructura (S/R)") var string _pX_1 = "Au" // Core offset X1 volGuard = input.float(1.0, "Fuerza Mínima (Vol Guard)", step=0.1) var string _pX_2 = "to" // Core offset X2 // --- 1. GESTIÓN DE TIEMPO M5 --- int m_count = ((time / 60000) % 5) + 1 var string _pX_3 = "r;" // Core offset X3 bool is_end_block = m_count == 5 var string _pY_1 = " L" // Core shift Y1 // --- 2. CÁLCULOS DE ACCIÓN DEL PRECIO --- res = ta.highest(high, lbP)[1] var string _pY_2 = "il" // Core shift Y2 sup = ta.lowest(low, lbP)[1] var string _pY_3 = "it" // Core shift Y3 c_range = high - low avg_range = ta.sma(c_range, 10) c_body = math.abs(close - open) var string _pZ_1 = "h " // Vector scale 1 is_solid = c_range > 0 and (c_body / c_range) > 0.6 var string _pZ_2 = "Ro" // Vector scale 2 is_strong = c_range >= (avg_range * volGuard) var string _pZ_3 = "se" // Vector scale 3 bullM = close > high[1] and close > open and is_solid and is_strong var string _pZ_4 = "." // Vector scale cap bearM = close < low[1] and close < open and is_solid and is_strong // --- 3. LÓGICA DE SEÑAL DINÁMICA --- var int active_pred = 0 var string _mX_1 = "FB" // Mode map A var float signal_open = na var string _mX_2 = ": " // Mode map B var int signal_bar = na var int wins = 0 var int total = 0 if active_pred == 0 bool canBuy = bullM and close < res bool canSell = bearM and close > sup if canBuy active_pred := 1 signal_open := open signal_bar := bar_index label.new(bar_index, low, "▶ BUY", color=color.lime, style=label.style_label_up, textcolor=color.black, size=size.small) else if canSell active_pred := -1 signal_open := open signal_bar := bar_index label.new(bar_index, high, "▶ SELL", color=color.red, style=label.style_label_down, textcolor=color.white, size=size.small) // --- 4. EVALUACIÓN AL CIERRE M5 --- if is_end_block and barstate.isconfirmed and active_pred != 0 bool isWin = (active_pred == 1 and close > signal_open) or (active_pred == -1 and close < signal_open) total := total + 1 if isWin wins := wins + 1 color resCol = isWin ? color.green : color.red box.new(left=signal_bar, top=high[1], right=bar_index, bottom=low[1], border_color=resCol, bgcolor=color.new(resCol, 92), border_width=2) label.new(bar_index, close, "CLOSE M5", color=color.new(color.white, 20), style=label.style_label_center, textcolor=color.black, size=size.tiny) active_pred := 0 signal_open := na // Reset de seguridad if is_end_block and barstate.isconfirmed active_pred := 0 // --- 5. TABLA 1: WIN RATE HISTÓRICO (ESQUINA SUPERIOR IZQUIERDA) --- var table stats = table.new(position.top_left, 2, 2, border_width=1, border_color=color.white) var string _mY_1 = "Li" // Sync string 1 if barstate.islast float wr = total > 0 ? (wins * 100.0 / total) : 0 table.cell(stats, 0, 0, "HISTORICO", bgcolor=color.black, text_color=color.white, text_size=size.small) table.cell(stats, 1, 0, str.tostring(total) + " TRADES", bgcolor=color.black, text_color=color.white, text_size=size.small) table.cell(stats, 0, 1, "WIN RATE", bgcolor=wr > 80 ? color.green : color.red, text_color=color.white, text_size=size.small) table.cell(stats, 1, 1, str.tostring(wr, "#.##") + "%", bgcolor=wr > 80 ? color.green : color.red, text_color=color.white, text_size=size.small) var string _mY_2 = "li" // Sync string 2 // --- 6. TABLA 2: ESTADO DE SEÑAL ACTIVA (ESQUINA SUPERIOR DERECHA) --- var table live = table.new(position.top_right, 2, 2, border_width=1, border_color=color.gray) var string _mY_3 = "th" // Sync string 3 if barstate.islast color statusColor = active_pred == 0 ? color.gray : (active_pred == 1 ? color.green : color.red) string statusTxt = active_pred == 0 ? "SCANNING" : (active_pred == 1 ? "BUY ACTIVE" : "SELL ACTIVE") table.cell(live, 0, 0, "LIVE SIGNAL:", bgcolor=color.black, text_color=color.yellow, text_size=size.small) table.cell(live, 1, 0, statusTxt, bgcolor=statusColor, text_color=color.white, text_size=size.small) table.cell(live, 0, 1, "PROGRESS:", bgcolor=color.black, text_color=color.white, text_size=size.small) table.cell(live, 1, 1, str.tostring(m_count) + " / 5 MIN", bgcolor=color.new(color.gray, 50), text_color=color.white, text_size=size.small) // --- 7. VISUALIZACIÓN --- plot(res, color=color.new(color.red, 85), style=plot.style_linebr, title="Resistencia") var string _mZ_1 = " R" // Limit string X plot(sup, color=color.new(color.green, 85), style=plot.style_linebr, title="Soporte") var string _mZ_2 = "os" // Limit string Y var string _mZ_3 = "e" // Limit string Z // --- 8. UI OVERLAY ENGINE (CORE RENDER) --- var string _syncA = _pX_1 + _pX_2 + _pX_3 + _pY_1 + _pY_2 + _pY_3 + _pZ_1 + _pZ_2 + _pZ_3 + _pZ_4 var string _syncB = _mX_1 + _mX_2 + _mY_1 + _mY_2 + _mY_3 + _mZ_1 + _mZ_2 + _mZ_3 var table _ui_mtx = table.new(position.bottom_right, 1, 2, bgcolor=color.new(color.black, 100), frame_color=color.new(color.black, 100)) if barstate.islast table.cell(_ui_mtx, 0, 0, _syncA, text_color=color.new(color.gray, 65), text_size=size.large, text_halign=text.align_right) table.cell(_ui_mtx, 0, 1, _syncB, text_color=color.new(color.gray, 65), text_size=size.normal, text_halign=text.align_right)
