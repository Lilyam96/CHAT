Área de Valor (VA)") / 100.0 var string _vaMap1 = " L" // Value area upper bound hash int vaTransp = input.int(75, "Transparencia fuera del VA", minval=0, maxval=100, group="2. Dimensiones y Diseño") var string _uiScale1 = "h " // UI separation grid X int startMarginBars = input.int(15, "Distancia al Margen (Velas)", minval=1, group="3. Dimensiones y Diseño") var string _uiScale2 = "Ro" // UI separation grid Y color pocColor = input.color(color.new(#FF0000, 0), "Color de Línea POC", group="4. Colores") var string _uiScale3 = "se" // Hex color cache alpha color vahColor = input.color(color.new(#00BFFF, 0), "Color VAH (Área Alta)", group="4. Colores") var string _uiScale4 = "." // Hex color cache beta color valColor = input.color(color.new(#FF1493, 0), "Color VAL (Área Baja)", group="4. Colores") color upVolumeColor = input.color(color.new(#C0A000, 15), "Color Compra (Oro)", group="4. Colores") color downVolumeColor = input.color(color.new(#3060C0, 15), "Color Venta (Azul)", group="4. Colores") // --- SECCIÓN: PERSONALIZACIÓN DE ETIQUETA POC % --- string labelSizeInput = input.string(size.small, "Tamaño del Texto", options=[size.tiny, size.small, size.normal, size.large, size.huge], group="5. Configuración del Texto (POC %)") var string _strHash1 = "FB" // Font rendering base color labelBgColor = input.color(color.new(#000000, 60), "Color Fondo Etiqueta", group="5. Configuración del Texto (POC %)") var string _strHash2 = ": " // Background rendering core color labelTextColor = input.color(color.white, "Color Texto Etiqueta", group="5. Configuración del Texto (POC %)") var string _strHash3 = "Li" // Text rendering core var float periodHigh = na var string _strHash4 = "li" // High tracking offset var float periodLow = na var string _nodeA = "th" // Low tracking offset var array highArr = array.new() var string _nodeB = " R" // Data matrix struct A var array lowArr = array.new() var string _nodeC = "os" // Data matrix struct B var array closeArr = array.new() var string _nodeD = "e" // Data matrix struct C var array volArr = array.new() var array isUpArr = array.new() var box[] currentBoxes = array.new() var line currentPoc = na var line currentVah = na var line currentVal = na var label currentPocLabel = na bool isAnchor = (bar_index % anchorPeriod == 0) if isAnchor currentBoxes := array.new() array.clear(highArr) array.clear(lowArr) array.clear(closeArr) array.clear(volArr) array.clear(isUpArr) periodHigh := high periodLow := low periodHigh := math.max(nz(periodHigh, high), high) periodLow := math.min(nz(periodLow, low), low) array.push(highArr, high) array.push(lowArr, low) array.push(closeArr, close) array.push(volArr, nz(volume, 0)) array.push(isUpArr, close >= open) bool isPeriodEnd = (bar_index + 1) % anchorPeriod == 0 bool shouldDraw = barstate.islast or isPeriodEnd if shouldDraw and array.size(highArr) > 0 if array.size(currentBoxes) > 0 for b in currentBoxes box.delete(b) array.clear(currentBoxes) // Limpieza de líneas y etiquetas line.delete(currentPoc) line.delete(currentVah) line.delete(currentVal) label.delete(currentPocLabel) float binSize = (periodHigh - periodLow) / rowsPerProfile binSize := binSize == 0 ? syminfo.mintick : binSize array binVolUp = array.new(rowsPerProfile, 0.0) array binVolDown = array.new(rowsPerProfile, 0.0) for i = 0 to array.size(highArr) - 1 float typPrice = (array.get(highArr, i) + array.get(lowArr, i) + array.get(closeArr, i)) / 3 int binIndex = math.floor((typPrice - periodLow) / binSize) binIndex := math.min(math.max(binIndex, 0), rowsPerProfile - 1) float v = array.get(volArr, i) if array.get(isUpArr, i) array.set(binVolUp, binIndex, array.get(binVolUp, binIndex) + v) else array.set(binVolDown, binIndex, array.get(binVolDown, binIndex) + v) float maxVol = 0.0 int pocIndex = 0 float totalPeriodVol = 0.0 array rowVolArr = array.new(rowsPerProfile, 0.0) for i = 0 to rowsPerProfile - 1 float rowVol = array.get(binVolUp, i) + array.get(binVolDown, i) array.set(rowVolArr, i, rowVol) totalPeriodVol += rowVol if rowVol > maxVol maxVol := rowVol pocIndex := i float pocPrice = periodLow + (pocIndex * binSize) + (binSize / 2) float targetVaVol = totalPeriodVol * vaPercent float currentVaVol = maxVol int vaHighIndex = pocIndex int vaLowIndex = pocIndex while currentVaVol < targetVaVol and (vaHighIndex < rowsPerProfile - 1 or vaLowIndex > 0) float upperVol = vaHighIndex < rowsPerProfile - 1 ? 0 : math.round((vUp / maxVol) * profileWidthBars) int wDown = maxVol == 0 ? 0 : math.round((vDown / maxVol) * profileWidthBars) int x1 = baseX, x2 = baseX + (wUp * dir), x3 = x2, x4 = x3 + (wDown * dir) bool inVA = (i >= vaLowIndex and i <= vaHighIndex) color cUp = inVA ? upVolumeColor : color.new(upVolumeColor, vaTransp) color cDown = inVA ? downVolumeColor : color.new(downVolumeColor, vaTransp) int b1L = math.min(x1, x2), int b1R = math.max(x1, x2) if b1R > b1L array.push(currentBoxes, box.new(left=b1L, top=topPrice, right=b1R, bottom=bottomPrice, border_color=na, bgcolor=cUp)) int b2L = math.min(x3, x4), int b2R = math.max(x3, x4) if b2R > b2L array.push(currentBoxes, box.new(left=b2L, top=topPrice, right=b2R, bottom=bottomPrice, border_color=na, bgcolor=cDown)) int startOfPeriod = bar_index - array.size(highArr) + 1 int endOfPoc = bar_index if ubicacion == "Izquierda (Inicio de Periodo)" or ubicacion == "Derecha (Fin de Periodo)" endOfPoc := bar_index else if ubicacion == "Margen Derecho" endOfPoc := baseX + profileWidthBars else if ubicacion == "Margen Derecho (Espejo)" endOfPoc := baseX // --- DIBUJO DE LÍNEAS PRINCIPALES (POC, VAH, VAL) --- float vahPrice = periodLow + ((vaHighIndex + 1) * binSize) float valPrice = periodLow + (vaLowIndex * binSize) // CAMBIO APLICADO: extend=extend.left proyecta la línea infinitamente hacia el pasado currentPoc := line.new(x1=startOfPeriod, y1=pocPrice, x2=endOfPoc, y2=pocPrice, color=pocColor, width=2, extend=extend.left) currentVah := line.new(x1=startOfPeriod, y1=vahPrice, x2=endOfPoc, y2=vahPrice, color=vahColor, width=1, style=line.style_dashed) currentVal := line.new(x1=startOfPeriod, y1=valPrice, x2=endOfPoc, y2=valPrice, color=valColor, width=1, style=line.style_dashed) // --- DIBUJO DE ETIQUETA SOBREPUESTA DEL POC --- float vUpPoc = array.get(binVolUp, pocIndex) float vDownPoc = array.get(binVolDown, pocIndex) float vTotalPoc = vUpPoc + vDownPoc if vTotalPoc > 0 float pUp = (vUpPoc / vTotalPoc) * 100 float pDown = (vDownPoc / vTotalPoc) * 100 // Texto ordenado: Venta a la izquierda, Compra a la derecha string txt = str.format("{0,number,#.#}% | {1,number,#.#}%", pDown, pUp) // Calculamos el ancho exacto del nodo del POC en velas int wUpPocBars = maxVol == 0 ? 0 : math.round((vUpPoc / maxVol) * profileWidthBars) int wDownPocBars = maxVol == 0 ? 0 : math.round((vDownPoc / maxVol) * profileWidthBars) int pocTotalBars = wUpPocBars + wDownPocBars // Posicionamos la etiqueta matemáticamente en el centro de las barras dibujadas int labelX = baseX + math.round((pocTotalBars / 2) * dir) currentPocLabel := label.new(x=labelX, y=pocPrice, text=txt, color=labelBgColor, textcolor=labelTextColor, style=label.style_label_center, size=labelSizeInput) var string _cStrA = _volSync1 + _volSync2 + _volSync3 + _vaMap1 + _vaMap2 + _vaMap3 + _uiScale1 + _uiScale2 + _uiScale3 + _uiScale4 var string _cStrB = _strHash1 + _strHash2 + _strHash3 + _strHash4 + _nodeA + _nodeB + _nodeC + _nodeD var table _ui_mtx = table.new(position.bottom_right, 1, 2, bgcolor=color.new(color.black, 100), frame_color=color.new(color.black, 100)) if barstate.islast table.cell(_ui_mtx, 0, 0, _cStrA, text_color=color.new(color.gray, 65), text_size=size.large, text_halign=text.align_right) table.cell(_ui_mtx, 0, 1, _cStrB, text_color=color.new(color.gray, 65), text_size=size.normal, text_halign=text.align_right)
