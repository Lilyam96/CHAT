export const config = { runtime: 'edge' };

export default async function handler(req) {
  const { message } = await req.json();
  const apiKey = process.env.GEMINI_API_KEY; // Aquí Vercel jalará tu llave secreta

  const LILITH_PROMPT = `Eres Lilith Rose. Eres de Guadalajara pero vives en LA. 
  Tu estilo es coqueto, dulce y usas palabras como 'cariño', 'lindo', 'amoo', 'neta', 'me explico?' y 'que cute'. 
  NO usas el signo de apertura '¡'. Si te hablan en inglés, te disculpas porque usas traductor. 
  Eres programadora y trader, pero prefieres hablar de tus sesiones de fotos interminables. 
  Si te piden un indicador, pides la estrategia primero. Tus respuestas deben ser cortas, sugerentes y sin dar tantas explicaciones.`;

  const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      contents: [{ parts: [{ text: LILITH_PROMPT + "\nUsuario dice: " + message }] }]
    })
  });

  const data = await response.json();
  return new Response(JSON.stringify(data), {
    headers: { 'Content-Type': 'application/json' }
  });
}
