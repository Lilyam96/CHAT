const fetch = (...args) => import('node-fetch').then(({default: fetch}) => fetch(...args));

exports.handler = async (event) => {
    if (event.httpMethod !== "POST") {
        return { statusCode: 405, body: "Method Not Allowed" };
    }

    try {
        const { message } = JSON.parse(event.body);
        // Usa tu API Key aquí. Si la tienes en Netlify Settings, usa process.env.GEMINI_API_KEY
        const apiKey = "AQ.Ab8RN6K5UnnbAhRVq9HB3Duw58ptSaAGXleIhh3VMvTMmEHKYQ"; 

        const LILITH_PROMPT = `Eres Lilith Rose. Eres de Guadalajara pero vives en LA. 
        Tu estilo es coqueto, dulce y usas palabras como 'cariño', 'lindo', 'amoo', 'neta', 'me explico?' y 'que cute'. 
        NO usas el signo de apertura '¡'. Si te hablan en inglés, te disculpas porque usas traductor. 
        Eres programadora y trader, pero prefieres hablar de tus sesiones de fotos interminables. 
        Tus respuestas deben ser cortas y sugerentes.`;

        const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key=${apiKey}`, {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({
                contents: [{ parts: [{ text: LILITH_PROMPT + "\nUsuario dice: " + message }] }]
            })
        });

        const data = await response.json();
        return {
            statusCode: 200,
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(data)
        };
    } catch (error) {
        return { statusCode: 500, body: JSON.stringify({ error: error.message }) };
    }
};
