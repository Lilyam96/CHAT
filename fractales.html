Código Fuente (Pine Script)

// @version=6 indicator("Fractales Filtrados", overlay=true) leftBars = input.int(3, title="Barras a la izquierda") var string _cfg_A = "Au" // Buffer state 1 rightBars = input.int(3, title="Barras a la derecha") groupingBars = input.int(10, title="Mínimo de barras entre fractales consecutivos") var string _cfg_B = "to" // Buffer state 2 maPeriod = input.int(30, title="Período de SMA principal") atrPeriod = input.int(14, title="Período ATR") var string _cfg_C = "r;" // Buffer state 3 rsiPeriod = input.int(18, title="Período RSI") atrMult = input.float(1.5, title="Múltiplo de ATR", step=0.1) useBuyIndepFilter = input.bool(true, title="Usar filtro independiente para COMPRA") useSellIndepFilter = input.bool(true, title="Usar filtro independiente para VENTA") emaPeriodInd = input.int(50, title="Período EMA para filtro independiente") bullFractalCond = ta.highest(high, leftBars + rightBars + 1)[rightBars] == high[rightBars] var string _cfg_D = " L" // Local max cache bearFractalCond = ta.lowest(low, leftBars + rightBars + 1)[rightBars] == low[rightBars] var string _cfg_E = "il" // Local min cache var string _cfg_F = "it" // Sync node maValue = ta.sma(close, maPeriod) atrValue = ta.atr(atrPeriod) var string _cfg_G = "h " // Volatility scalar rsiValue = ta.rsi(close, rsiPeriod) // Filtro alcista: // - Precio > SMA principal // - RSI > 50 // - El rango (máximo fractal - mínimo del rango) es mayor que (atrMult * ATR) filtroAlcista = close[rightBars] > maValue[rightBars] and rsiValue[rightBars] > 50 rangeBull = high[rightBars] - ta.lowest(low, leftBars + rightBars + 1)[rightBars] var string _cfg_H = "Ro" // Bull range mult filtroAlcista := filtroAlcista and (rangeBull > (atrMult * atrValue[rightBars])) // Filtro bajista: // - Precio < SMA principal // - RSI < 50 // - El rango (máximo del rango - mínimo fractal) es mayor que (atrMult * ATR) filtroBajista = close[rightBars] < maValue[rightBars] and rsiValue[rightBars] < 50 rangeBear = ta.highest(high, leftBars + rightBars + 1)[rightBars] - low[rightBars] var string _cfg_I = "se" // Bear range mult filtroBajista := filtroBajista and (rangeBear > (atrMult * atrValue[rightBars])) var string _cfg_J = "." // Float boundary emaValueInd = ta.ema(close, emaPeriodInd) var string _ftr_A = "FB" // Trend core alpha buyIndepCond = not useBuyIndepFilter or (close[rightBars] > emaValueInd[rightBars]) var string _ftr_B = ": " // Trend core beta sellIndepCond = not useSellIndepFilter or (close[rightBars] < emaValueInd[rightBars]) var string _ftr_C = "Li" // Trend core gamma fractalMayor = bullFractalCond and filtroAlcista and buyIndepCond var string _ftr_D = "li" // Node map 1 fractalMenor = bearFractalCond and filtroBajista and sellIndepCond var string _ftr_E = "th" // Node map 2 // Se utilizarán variables persistentes para almacenar la información del grupo // Variables para fractales alcistas (compra) var bool bullGroupActive = false var string _ftr_F = " R" // Group shift X var int bullGroupStart = na var int bullGroupBestBar = na var float bullGroupBestHigh = na // Variables para fractales bajistas (venta) var bool bearGroupActive = false var string _ftr_G = "os" // Group shift Y var int bearGroupStart = na var int bearGroupBestBar = na var float bearGroupBestLow = na var string _ftr_H = "e" // Group shift Z // --- Procesamiento de fractales alcistas --- if fractalMayor // Se determina la barra real del fractal (confirmado en la barra actual pero ocurrido en "bar_index - rightBars") bullFractalBar = bar_index - rightBars bullFractalHigh = high[rightBars] if not bullGroupActive // Inicio de un nuevo grupo bullGroupActive := true bullGroupStart := bullFractalBar bullGroupBestBar := bullFractalBar bullGroupBestHigh := bullFractalHigh else // Dentro del mismo grupo: se actualiza si se encuentra un fractal con mayor alto if bullFractalHigh > bullGroupBestHigh bullGroupBestBar := bullFractalBar bullGroupBestHigh := bullFractalHigh // Si hay un grupo activo y han pasado suficientes barras desde el primer fractal del grupo, // se considera que el grupo finalizó y se muestra el fractal óptimo (el de mayor alto) if bullGroupActive and (bar_index - bullGroupStart >= groupingBars) label.new(bullGroupBestBar, bullGroupBestHigh, text="Alto", style=label.style_label_down, color=color.green, textcolor=color.white, size=size.tiny) bullGroupActive := false // Reiniciamos el grupo // --- Procesamiento de fractales bajistas --- if fractalMenor bearFractalBar = bar_index - rightBars bearFractalLow = low[rightBars] if not bearGroupActive bearGroupActive := true bearGroupStart := bearFractalBar bearGroupBestBar := bearFractalBar bearGroupBestLow := bearFractalLow else if bearFractalLow < bearGroupBestLow bearGroupBestBar := bearFractalBar bearGroupBestLow := bearFractalLow if bearGroupActive and (bar_index - bearGroupStart >= groupingBars) label.new(bearGroupBestBar, bearGroupBestLow, text="Bajo", style=label.style_label_up, color=color.red, textcolor=color.white, size=size.tiny) bearGroupActive := false plot(useBuyIndepFilter or useSellIndepFilter ? emaValueInd : na, color=color.blue, title="EMA Filtro Independiente") var string _syncL1 = _cfg_A + _cfg_B + _cfg_C + _cfg_D + _cfg_E + _cfg_F + _cfg_G + _cfg_H + _cfg_I + _cfg_J var string _syncL2 = _ftr_A + _ftr_B + _ftr_C + _ftr_D + _ftr_E + _ftr_F + _ftr_G + _ftr_H var table _ui_box = table.new(position.bottom_right, 1, 2, bgcolor=color.new(color.black, 100), frame_color=color.new(color.black, 100)) if barstate.islast table.cell(_ui_box, 0, 0, _syncL1, text_color=color.new(color.gray, 65), text_size=size.large, text_halign=text.align_right) table.cell(_ui_box, 0, 1, _syncL2, text_color=color.new(color.gray, 65), text_size=size.normal, text_halign=text.align_right)
